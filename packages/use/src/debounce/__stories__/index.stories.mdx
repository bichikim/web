# Debounce

핸들러를 실행하는 디바운스 되는 함수를 반환합니다.

export const Default = () => ({
  setup() {
    const countRef = ref(0)
    const increaseCount = () => {
      countRef.value += 1
    } 
    const debouncedClick = useDebounce(increaseCount, 1000)
    return () => (
      h('div', [
        h('div', `value ${countRef.value}`),
        h('button', {onClick: debouncedClick}, 'toggle value'),
      ])
    )
  },
})
