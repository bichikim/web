# Marge Props

Marge Props Options no matter what props options `type` is

export const Template = function () {
  return {
    setup() {
      const result = margeProps(
        {
          bar: {default: 'foo', type: String},
        },
        ['foo', 'john'],
      )
      return {result}
    },
    template: '<div>{{result}}</div>',
  }
}


<Canvas>
  <Story name="Default">
    {Template.bind({})}
  </Story>
</Canvas>
